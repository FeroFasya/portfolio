<script>
  const terminalInput = document.getElementById('terminal-input');
  const terminalOutput = document.getElementById('terminal-output');

  const commands = {
    help: () => `Available commands:\n- help\n- show skills\n- about me\n- projects\n- contact\n- clear\n- exit`,
    'show skills': () => `Skills:\nHTML/CSS: 60%\nVideo Editing: 65%\nBackend Development: 20%`,
    'about me': () => {
      window.location.href = 'about.html';
      return 'Redirecting to About Me...';
    },
    projects: () => {
      window.location.href = 'projects.html';
      return 'Opening Projects...';
    },
    contact: () => {
      window.location.href = 'kontak.html';
      return 'Redirecting to Contact Info...';
    },
    clear: () => {
      terminalOutput.innerText = '';
      return '';
    },
    exit: () => 'See you in the real world, coder~'
  };

  terminalInput.addEventListener('keydown', function (e) {
    if (e.key === 'Enter') {
      const input = terminalInput.value.trim();
      terminalInput.value = '';
      const response = commands[input] ? commands[input]() : `Command not found: ${input}`;
      if (response) {
        terminalOutput.innerText += `> ${input}\n${response}\n\n`;
      }
    }
  });
</script>
